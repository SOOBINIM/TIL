const deepObject = {
	state : {
		information : {
			name : “쫑아”,
			habit : [”긁어달라하기”,”꼬리흔들기”,”짓기”]
		}
	},
age : 14
};

// 2번에 나눠서 불러오는 방법
const {name, habit} = deepObject.state.information;
const {age} = deepObject

// 1번에 불러오는 방법
const = {
	state : {
		information : {name, habit}
	},
age
} = deepObject;

	name, // name : name
	habit, // habit : habit
	age, // age : age
}

console.log(extracted)
